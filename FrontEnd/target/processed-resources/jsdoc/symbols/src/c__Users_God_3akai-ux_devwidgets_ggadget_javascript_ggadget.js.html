 18 /*global $, get_cookie, Config */
 19 
 20 var sakai = sakai || {};
 21 
 22 /**
 23  * @name sakai.ggadget
 24  *
 25  * @class ggadget
 26  *
 27  * @description
 28  * Initialize the ggadget widget
 29  *
 30  * @version 0.0.1
 31  * @param {String} tuid Unique id of the widget
 32  * @param {Boolean} showSettings Show the settings of the widget or not
 33  */
 34 sakai.ggadget = function(tuid, showSettings){
 35 
 36 
 37     /////////////////////////////
 38     // Configuration variables //
 39     /////////////////////////////
 40 
 41     var rootel = $("#" + tuid);
 42     var json = false;
 43     var isAdvancedSettingsVisible = false;
 44 
 45     // Default values
 46     var defaultWidth = 600;
 47     var defaultWidthUnit = "px";
 48     var defaultHeight = 400;
 49 
 50     // Links and labels
 51     var remotecontent = "#remotecontent";
 52     var remotecontentSettings = remotecontent + "_settings";
 53     var remotecontentSettingsAdvanced = remotecontentSettings + "_advanced";
 54     var remotecontentSettingsAdvancedDown = remotecontentSettingsAdvanced + "_down";
 55     var remotecontentSettingsAdvancedToggleSettings = remotecontentSettingsAdvanced + "_toggle_settings";
 56     var remotecontentSettingsAdvancedUp = remotecontentSettingsAdvanced + "_up";
 57     var remotecontentSettingsBorders = remotecontentSettings + "_borders";
 58     var remotecontentSettingsCancel = remotecontentSettings + "_cancel";
 59     var remotecontentSettingsColorContainer = remotecontentSettings + "_color_container";
 60     var remotecontentSettingsHeight = remotecontentSettings + "_height";
 61     var remotecontentSettingsInsert = remotecontentSettings + "_insert";
 62     var remotecontentSettingsPreview = remotecontentSettings + "_preview";
 63     var remotecontentSettingsPreviewFrame = remotecontentSettingsPreview + "_frame";
 64     var remotecontentSettingsUrl = remotecontentSettings + "_url";
 65     var remotecontentSettingsUrlError = remotecontentSettingsUrl + "_error";
 66     var remotecontentSettingsUrlErrorTitle = remotecontentSettingsUrlError + "_title";
 67     var remotecontentSettingsWidth = remotecontentSettings + "_width";
 68 
 69     // Containers
 70     var remotecontentMainContainer = remotecontent + "_main_container";
 71 
 72     // Classes
 73     var remotecontentSettingsWidthUnitClass = ".remotecontent_settings_width_unit";
 74     var remotecontentSettingsWidthUnitSelectedClass = "remotecontent_settings_width_unit_selected";
 75 
 76     // Templates
 77     var $remotecontentSettingsColorContainerTemplate = $("#remotecontent_settings_color_container_template", rootel);
 78     var $remotecontentSettingsTemplate = $("#remotecontent_settings_template", rootel);
 79     var $remotecontentSettingsPreviewTemplate = $("#remotecontent_settings_preview_template", rootel);
 80 
 81     ///////////////////////
 82     // Utility functions //
 83     ///////////////////////
 84 
 85     /**
 86      * Check if the value is a decimal or not
 87      * @param {Object} value Value that needs to be checked
 88      * @return {Boolean}
 89      *     true: is a decimal
 90      *     false: is not a decimal
 91      */
 92     var isDecimal = function(value){
 93         return (/^\d+$/).test(value);
 94     };
 95 
 96     /**
 97      * Check if the input url is in fact an url or not
 98      * @param {String} url Url that needs to be tested
 99      * @return {Boolean}
100      *     true: is an url
101      *     false: is not an url
102      */
103     var isUrl = function(url){
104         return (/^http\:\/\/|^https\:\/\//i).test(url);
105     };
106 
107     /**
108      * Called when the data has been saved to the JCR.
109      */
110     var savedDataToJCR = function(){
111         sakai.api.Widgets.Container.informFinish(tuid, "ggadget");
112     };
113 
114 
115     //////////////////////
116     // Render functions //
117     //////////////////////
118 
119     /**
120      * Render the iframe for the widget in settings mode
121      * @param {Boolean} complete Render the preview completely or only adjust values
122      */
123     var renderIframeSettings = function(complete){
124         if (complete) {
125             // We create this object to render the iframe with the default height, width and widthunit
126             var jsonDefaultSize = {};
127             jsonDefaultSize = json;
128             jsonDefaultSize.width = defaultWidth;
129             jsonDefaultSize.width_unit = defaultWidthUnit;
130             jsonDefaultSize.height = defaultHeight;
131             $(remotecontentSettingsPreview).html($.TemplateRenderer($remotecontentSettingsPreviewTemplate, json, null, false));
132         }
133         else {
134             $(remotecontentSettingsPreviewFrame).attr("style", "border: " + json.border_size + "px #" + json.border_color + " solid");
135         }
136     };
137 
138     /**
139      * Render the iframe for the widget
140      */
141     var renderIframe = function(){
142         if (json) {
143             $(remotecontentMainContainer, rootel).html($.TemplateRenderer($remotecontentSettingsPreviewTemplate, json, null, false));
144 
145             // SAKIII-314 We need to show the container, otherwise the second item won't be shown.
146             $(remotecontentMainContainer, rootel).show();
147         }
148     };
149 
150     /**
151      * Render the html of the remotecontentsettings
152      */
153     var renderRemoteContentSettings = function(){
154         if (json) {
155             $(remotecontentSettings).html($.TemplateRenderer($remotecontentSettingsTemplate, json));
156         }
157     };
158 
159     /**
160      * Render the color container
161      */
162     var renderColorContainer = function(){
163         if (json) {
164             $(remotecontentSettingsColorContainer).html($.TemplateRenderer($remotecontentSettingsColorContainerTemplate, json));
165         }
166     };
167 
168 
169     //////////////////////
170     // Global functions //
171     //////////////////////
172 
173     /**
174      * Display the iframe in normal mode
175      * @param {Object} parameters JSON object that contains the necessary information for the iframe
176      */
177     var displayRemoteContent = function(parameters){
178         json = parameters;
179         renderIframe();
180         $("iframe").ready(function() {
181             $("iframe iframe").attr("style", "overflow:auto");
182         });
183 
184 
185     };
186 
187     /**
188      * Save the remotecontent to the jcr
189      */
190     var saveRemoteContent = function(){
191         if (json.url !== "") {
192             sakai.api.Widgets.saveWidgetData(tuid, json, savedDataToJCR);
193         }
194         else {
195             sakai.api.Util.notification.show($(remotecontentSettingsUrlErrorTitle).html(), $(remotecontentSettingsUrlError).html());
196         }
197     };
198 
199     /**
200      * Change the direction (up/down) of the arrow for the advanced settings
201      */
202     var changeAdvancedSettingsArrow = function(){
203         if (isAdvancedSettingsVisible) {
204             $(remotecontentSettingsAdvancedDown, rootel).hide();
205             $(remotecontentSettingsAdvancedUp, rootel).show();
206         }
207         else {
208             $(remotecontentSettingsAdvancedUp, rootel).hide();
209             $(remotecontentSettingsAdvancedDown, rootel).show();
210         }
211     };
212 
213 
214     //////////////
215     // Bindings //
216     //////////////
217 
218     /*
219      * Add binding to the color boxes
220      */
221     var addColorBinding = function(){
222         $(".remotecontent_settings_color").click(function(){
223             json.border_color = $(this).attr("id").split("_")[$(this).attr("id").split("_").length - 1];
224             renderIframeSettings(false);
225             renderColorContainer();
226             addColorBinding();
227         });
228     };
229 
230     /**
231      * previewGadget
232      * Executes the preview for the gadget based on the embed code
233      */
234     var previewGadget = function() {
235         // get the src attribute of the embed script tag, and define a html render rather than JS
236             var urlValue = $($(remotecontentSettingsUrl).val()).attr("src").replace("output=js", "output=html");
237 
238             // Get size of the gadget from the embed code
239             var rawParams = urlValue.split("&");
240             for (var i = 0, il = rawParams.length; i < il; i++) {
241                 var kvpair = rawParams[i].split("=");
242                 if (kvpair[0] === "w") {
243                     json.width = kvpair[1];
244                 }
245                 if (kvpair[0] === "h") {
246                     json.width = kvpair[1];
247                 }
248             }
249 
250             // Set width and height values in settings
251             if (isDecimal(json.width)) {
252                 $(remotecontentSettingsWidth).val(json.width);
253             }
254 
255             if (isDecimal(json.height)) {
256                 $(remotecontentSettingsHeight).val(json.height);
257             }
258 
259             if (urlValue !== "") {
260                 // Check if someone already wrote http inside the url
261                 if (!isUrl(urlValue)) {
262                     urlValue = 'http://' + urlValue;
263                 }
264                 json.url = urlValue;
265 
266                 renderIframeSettings(true);
267             }
268     };
269 
270 
271     /*
272      * Add binding to all the elements
273      */
274     var addBinding = function(){
275 
276         // Change the url for the iFrame
277         $(remotecontentSettingsUrl).change(function(){
278 
279             previewGadget();
280 
281         });
282 
283         // Preview gadget
284         $(".ggadget_preview_button", rootel).bind("click", function() {
285 
286             if ($(remotecontentSettingsUrl).val() !== "") {
287                 previewGadget();
288             }
289            
290             return false;
291 
292         });
293 
294         // Change the iframe width
295         $(remotecontentSettingsWidth).change(function(){
296             var widthValue = $(remotecontentSettingsWidth).val();
297 
298             if (isDecimal(widthValue)) {
299                 json.width = widthValue;
300             }
301             renderIframeSettings(false);
302         });
303 
304         // Change the iframe height
305         $(remotecontentSettingsHeight).change(function(){
306             var heightValue = $(remotecontentSettingsHeight).val();
307 
308             if (isDecimal(heightValue)) {
309                 json.height = heightValue;
310             }
311             renderIframeSettings(false);
312         });
313 
314         // Change the border width
315         $(remotecontentSettingsBorders).change(function(){
316             var borderValue = $(remotecontentSettingsBorders).val();
317             if (isDecimal(borderValue)) {
318                 json.border_size = borderValue;
319                 renderIframeSettings(false);
320             }
321         });
322 
323         // Toggle the advanced view
324         $(remotecontentSettingsAdvancedToggleSettings).click(function(){
325             $("#remotecontent_settings_advanced", rootel).toggle();
326             isAdvancedSettingsVisible = !isAdvancedSettingsVisible;
327             changeAdvancedSettingsArrow();
328         });
329 
330         // When you click on one of the width units (px or percentage)
331         $(remotecontentSettingsWidthUnitClass).click(function(){
332             var widthUnitValue = $(this).attr("id").split("_")[$(this).attr("id").split("_").length - 1];
333             if (widthUnitValue === "px") {
334                 json.width_unit = widthUnitValue;
335             }
336             else {
337                 json.width_unit = "%";
338             }
339             $(remotecontentSettingsWidthUnitClass).removeClass(remotecontentSettingsWidthUnitSelectedClass);
340             $(this).addClass(remotecontentSettingsWidthUnitSelectedClass);
341             renderIframeSettings(false);
342         });
343 
344         // When you push the save button..
345         $(remotecontentSettingsInsert).click(function(){
346             saveRemoteContent();
347         });
348 
349         // Cancel it
350         $(remotecontentSettingsCancel).click(function(){
351             sakai.api.Widgets.Container.informCancel(tuid, "ggadget");
352         });
353 
354         addColorBinding();
355     };
356 
357 
358     ///////////////////////
359     // Initial functions //
360     ///////////////////////
361 
362     /**
363      * Function that fills in the input fields in the settings tab.
364      * @param {Object} parameters A JSON object that contains the necessary information.
365      * @param {Boolean} exists Does there exist a previous remotecontent
366      */
367     var displaySettings = function(parameters, exists){
368         if (exists && parameters.url) {
369             json = parameters;
370         }
371         else {
372             json = {
373                 border_size: 0,
374                 border_color: "cccccc",
375                 height: defaultHeight,
376                 url: "",
377                 width: defaultWidth,
378                 width_unit: defaultWidthUnit
379             };
380         }
381         renderRemoteContentSettings();
382         renderIframeSettings(true);
383         renderColorContainer();
384         addBinding(); // Add binding to the various elements
385         changeAdvancedSettingsArrow();
386         $(remotecontentSettings).show(); // Show the remotecontent settings
387         $(remotecontentSettingsUrl).focus();
388     };
389 
390     /*
391      * Is the widget in settings mode or not
392      */
393     if (showSettings) {
394         $(remotecontentMainContainer).hide();
395         $(remotecontentSettings).show();
396     }
397     else {
398         $(remotecontentSettings).hide();
399         $(remotecontentMainContainer).show();
400     }
401 
402     /**
403      * Will fetch the URL and other parameters from the JCR and according to which
404      * view we are in, fill in the settings or display an iframe.
405      */
406     var getRemoteContent = function(){
407 
408         sakai.api.Widgets.loadWidgetData(tuid, function(success, data){
409             if (success) {
410                 // Get a JSON string that contains the necessary information.
411                 var parameters = data;
412 
413                 if (showSettings) {
414                     displaySettings(parameters, true); // Fill in the settings page.
415                 }
416                 else {
417                     displayRemoteContent(parameters); // Show the frame
418                 }
419             }
420             else {
421                 if (showSettings) {
422                     // When the request isn't successful, it means that  there was no existing remotecontent
423                     // so we show the basic settings.
424                     displaySettings(null, false);
425                 }
426             }
427         });
428     };
429 
430     getRemoteContent();
431 };
432 
433 sakai.api.Widgets.widgetLoader.informOnLoad("ggadget");
