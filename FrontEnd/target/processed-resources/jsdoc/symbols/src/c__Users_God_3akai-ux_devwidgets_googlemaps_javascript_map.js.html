 19 // Init google map object with Cambridge location as the center position
 20 var map = new google.maps.Map(document.getElementById("googlemaps_map_canvas"), {
 21     zoom: 8,
 22     center: new google.maps.LatLng(52.2025441, 0.1312368),
 23     mapTypeControlOptions: { style: google.maps.MapTypeControlStyle.DROPDOWN_MENU },
 24     mapTypeId: google.maps.MapTypeId.ROADMAP,
 25     navigationControlOptions: { style: google.maps.NavigationControlStyle.ZOOM_PAN }
 26 });
 27 
 28 // Init google marker object with Cambridge location as the center position
 29 var marker = new google.maps.Marker({
 30     position: new google.maps.LatLng(52.2025441, 0.1312368),
 31     title: 'Point A',
 32     map: map,
 33     draggable: false
 34 });
 35 
 36 // Init a google maps infor window object to open a info window to show the location info
 37 var infoWindow = new google.maps.InfoWindow({
 38     content: ""
 39 });
 40 
 41 // Init a json object which inlcudes the basic properties of the google map
 42 var json = {
 43     "lat": map.getCenter().lat(),
 44     "lng": map.getCenter().lng(),
 45     "mapzoom": map.getZoom(),
 46     "mapinput": "",
 47     "mapsize": "",
 48     "maphtml": "Cambridge, UK"
 49 };
 50 
 51 /**
 52  * Update the content in infoWindow
 53  */
 54 function updateInfoWindow(html) {
 55     if (infoWindow) {
 56         infoWindow.close();
 57     }
 58 
 59     json.maphtml = html;
 60     infoWindow = new google.maps.InfoWindow({
 61         content: "<h4>" + html + "</h4>"
 62     });
 63     infoWindow.open(map, marker);
 64 }
 65 
 66 /**
 67  * Find the closest position near the marker's location
 68  * @param {Object} position of the closest location
 69  */
 70 function geocodePosition(position) {
 71     var geocoder = new google.maps.Geocoder();
 72     if (geocoder) {
 73         geocoder.geocode({
 74             latLng: position
 75         }, function(responses){
 76             if (responses && responses.length > 0) {
 77                 var html = responses[0].formatted_address;
 78                 updateInfoWindow(html);
 79             }
 80             else {
 81                 alert($("#map_no_address").html());
 82             }
 83         });
 84     }
 85 }
 86 
 87 /**
 88  * Find the closest position near the marker's location
 89  * @param {Object} address the position of the closest location
 90  */
 91 function geocodeAddress(address) {
 92     var geocoder = new google.maps.Geocoder();
 93     if (geocoder) {
 94         geocoder.geocode({
 95             "address": address
 96         }, function(results, status){
 97             if (status == google.maps.GeocoderStatus.OK) {
 98                 var position = results[0].geometry.location;
 99                 map.setCenter(position);
100                 marker.setMap(map);
101                 marker.setPosition(position);
102                 var html = address;
103                 updateInfoWindow(html);
104             }
105             else {
106                 alert($("#map_no_address").html());
107             }
108         });
109     }
110 }
111 
112 /**
113  * Get the json object
114  */
115 var getJSON = function() {
116     // use marker rather than map because 
117     // map use 14 decimal places while marker uses only 6 decimal places 
118     // causing marker to appear on wrong spot
119     json.lat = marker.getPosition().lat();
120     json.lng = marker.getPosition().lng();
121     json.mapzoom = map.getZoom();
122     return json;
123 };
124 
125 /**
126  * Search the corresponded city between the target location
127  * @param {string} keyword the target location
128  * @param {string} The region where Google will perform the search
129  */
130 function search(keyword, region) {
131 
132     // If region is provided, attach it ot the search query
133     // TO DO: This later will need to be moved to the Google API
134     if (region !== "") {
135         geocodeAddress(keyword+", "+region);
136         json.mapinput = keyword+", "+region;
137     } else {
138         geocodeAddress(keyword);
139         json.mapinput = keyword;
140     }
141 }
142 
143 /**
144  * Set the map initial property (center an zoom)
145  * @param {Object} jsonTarget the json object includes map's properties
146  */
147 function setMap(jsonTarget) {
148     json = jsonTarget;
149 
150     var latLng = new google.maps.LatLng(json.lat, json.lng);
151     map.setCenter(latLng);
152     map.setZoom(json.mapzoom);
153     marker.setPosition(latLng);
154 
155     // Init the infoWindow object and decide to open it or not
156     if (json.maphtml) {
157         updateInfoWindow(json.maphtml);
158     }
159 }

161 /**
162  * Init the listeners of marker
163  */
164 function init() {
165 
166     //alert(new google.maps.LatLng(google.loader.ClientLocation.latitude, google.loader.ClientLocation.longitude))
167     google.maps.event.addListener(marker, "click", function() {
168         geocodePosition(marker.getPosition());
169     });
170 }
