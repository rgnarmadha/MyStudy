 18 /*global $, Config, jQuery, sakai */

 20 /**
 21  * @name sakai.mycontent
 22  *
 23  * @class mycontent
 24  *
 25  * @description
 26  * The 'My Content' widget shows the five most recently updated
 27  * content items the user manages.
 28  *
 29  * @version 0.0.1
 30  * @param {String} tuid Unique id of the widget
 31  * @param {Boolean} showSettings Show the settings of the widget or not
 32  */
 33 sakai.mycontent = function(tuid, showSettings) {
 34 
 35 
 36     /////////////////////////////
 37     // Configuration variables //
 38     /////////////////////////////
 39 
 40     // DOM identifiers
 41     var rootel = $("#" + tuid);
 42     var uploadLink = "#upload_link";
 43     var fileuploadContainer = "#fileupload_container";
 44     var noContentMsg = "#mycontent_nocontent";
 45     var dataErrorMsg = "#mycontent_data_error";
 46     var contentList = "#mycontent_list";
 47     var listTemplate = "#mycontent_list_template";
 48     var ellipsisContainer = ".mycontent_ellipsis_container";
 49 
 50 
 51     ///////////////////////
 52     // Utility functions //
 53     ///////////////////////
 54 
 55     /**
 56      * Parses an individual JSON search result (returned from the
 57      * /var/search/pool/me/manager.json data feed) to be displayed in
 58      * mycontent.html.
 59      * @param {Object} result - individual result object from JSON data feed
 60      * @return {Object} object containing item.name, item.path, item.type (mimetype)
 61      *   and item.type_img_url (URL for mimetype icon) for the given result
 62      */
 63     var parseDataResult = function(result) {
 64         // initialize parsed item with default values
 65         var item = {
 66             name: result["sakai:pooled-content-file-name"],
 67             path: "/p/" + result["jcr:name"],
 68             type: sakai.config.MimeTypes.other.description,
 69             type_img_url: sakai.config.MimeTypes.other.URL,
 70             size: ""
 71         };
 72 
 73         // set the mimetype and corresponding image
 74         var type = result["jcr:content"]["jcr:mimeType"];
 75         if(sakai.config.MimeTypes[type]) {
 76             // we have a recognized file type - set the description and img URL
 77             item.type = sakai.config.MimeTypes[type].description;
 78             item.type_img_url = sakai.config.MimeTypes[type].URL;
 79         }
 80 
 81         // set file name without the extension
 82         // be aware that links don't have an extension
 83         var lastDotIndex = result["sakai:pooled-content-file-name"].lastIndexOf(".");
 84         if(lastDotIndex !== -1) {
 85             if (type !== "x-sakai/link") {
 86                 // extension found
 87                 item.name = result["sakai:pooled-content-file-name"].slice(0, lastDotIndex);
 88             }
 89         }
 90 
 91         // set the file size
 92         var size = result["jcr:content"]["jcr:data"];
 93         if(size && size.indexOf("binary-length:") !== -1) {
 94             // the value exists and contains a binary length that we can turn
 95             // into a human readable file size
 96             item.size = "(" + sakai.api.Util.convertToHumanReadableFileSize(size.split(":")[1]) + ")";
 97         }
 98 
 99         return item;
100     };
101 
102     /**
103      * This AJAX callback function handles the search result data returned from
104      * /var/search/pool/me/manager.json.  If the call was successful, up to 5 of
105      * the most recently created files are presented to the user.
106      * @param {Object} success - indicates the status of the AJAX call
107      * @param {Object} data - JSON data from /var/search/pool/me/manager.json
108      * @return None
109      */
110     var handleContentData = function(success, data) {
111         if(success) {
112             // parse & render data
113             if(data.total < 1) {
114                 // user manages no content
115                 $(noContentMsg, rootel).show();
116             } else {
117                 // user manages content
118                 $(noContentMsg, rootel).hide();
119                 // build array of up to five items; reverse chronological order
120                 var contentjson = {
121                     items: []
122                 };
123                 for(var i = 0; i < data.total && i < 5; i++) {
124                     contentjson.items.push(parseDataResult(data.results[i]));
125                 }
126                 // pass the array to HTML view
127                 $(contentList, rootel).html($.TemplateRenderer($(listTemplate), contentjson));
128                 $(contentList, rootel).show();
129 
130                 // make sure the newly added content is properly styled with
131                 // threedots truncation
132                 $(ellipsisContainer).css("width", $(ellipsisContainer).width() + "px");
133                 $(ellipsisContainer, rootel).ThreeDots({
134                     max_rows: 1,
135                     text_span_class: "mycontent_ellipsis_text",
136                     e_span_class: "mycontent_e_span_class",
137                     whole_word: false,
138                     alt_text_t: true
139                 });
140             }
141         } else {
142             // display something useful to the user
143             $(dataErrorMsg, rootel).show();
144         }
145     };
146 
147 
148     ////////////////////
149     // Event Handlers //
150     ////////////////////
151 
152     // Clicking to upload content
153     $(uploadLink, rootel).click(function(ev) {
154         $(window).trigger("sakai-fileupload-init");
155         return false;
156     });
157 
158     // Listen for sakai-fileupload-complete event (from the fileupload widget)
159     // to refresh this widget's file listing
160     $(window).bind("sakai-fileupload-complete", function() {
161         init();
162     });
163 
164 
165     /////////////////////////////
166     // Initialization function //
167     /////////////////////////////
168 
169     /**
170      * Initiates fetching content data to be displayed in the My Content widget
171      * @return None
172      */
173     var init = function() {
174         // get list of content items
175         sakai.api.Server.loadJSON("/var/search/pool/me/manager.1.json",
176             handleContentData, {
177                 "q": "*",
178                 "sortOn": "jcr:created",
179                 "sortOrder": "descending",
180                 "page": "0",
181                 "items": "5"
182             }
183         );
184     };
185 
186     // run init() function when sakai.content object loads
187     init();
188 };
189 
190 sakai.api.Widgets.widgetLoader.informOnLoad("mycontent");
