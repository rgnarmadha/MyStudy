123     /**
124      * Parse a json string to a valid date
125      * @param {String} dateInput String of a date that needs to be parsed
126      * @returns {Date}
127      */
128     var parseDate = function(dateInput){
129         /** Get the date with the use of regular expressions */
130         if (dateInput !== null) {
131             /** Get the date with the use of regular expressions */
132             var match = /([0-9]{4})\-([0-9]{2})\-([0-9]{2}).([0-9]{2}):([0-9]{2}):([0-9]{2})/.exec(dateInput); // 2009-08-14T12:18:50
133             var d = new Date();
134             d.setYear(match[1]);
135             d.setMonth(match[2] - 1);
136             d.setDate(match[3]);
137             d.setHours(match[4]);
138             d.setMinutes(match[5]);
139             d.setSeconds(match[6]);
140             return d;
141         }
142         return null;
143 
144     };

146     /**
147      * returns how many years, months, days or hours since the dateinput
148      * @param {Date} date
149      */
150     var getTimeAgo = function(date){
151         if (date !== null) {
152             var currentDate = new Date();
153             var iTimeAgo = (currentDate - date) / (1000);
154             if (iTimeAgo < 60) {
155                 if (Math.floor(iTimeAgo) === 1) {
156                     return Math.floor(iTimeAgo) +" " + sakai.api.i18n.General.getValueForKey("SECOND");
157                 }
158                 return Math.floor(iTimeAgo) + " "+sakai.api.i18n.General.getValueForKey("SECONDS");
159             }
160             else
161                 if (iTimeAgo < 3600) {
162                     if (Math.floor(iTimeAgo / 60) === 1) {
163                         
164                         return Math.floor(iTimeAgo / 60) + " "+sakai.api.i18n.General.getValueForKey("MINUTE");
165                     }
166                     return Math.floor(iTimeAgo / 60) + " "+sakai.api.i18n.General.getValueForKey("MINUTES");
167                 }
168                 else
169                     if (iTimeAgo < (3600 * 60)) {
170                         if (Math.floor(iTimeAgo / (3600)) === 1) {
171                             return Math.floor(iTimeAgo / (3600)) + " "+sakai.api.i18n.General.getValueForKey("HOUR");
172                         }
173                         return Math.floor(iTimeAgo / (3600)) + " "+sakai.api.i18n.General.getValueForKey("HOURS");
174                     }
175                     else
176                         if (iTimeAgo < (3600 * 60 * 30)) {
177                             if (Math.floor(iTimeAgo / (3600 * 60)) === 1) {
178                                 return Math.floor(iTimeAgo / (3600 * 60)) + " "+sakai.api.i18n.General.getValueForKey("DAY");
179                             }
180                             return Math.floor(iTimeAgo / (3600 * 60)) + " "+sakai.api.i18n.General.getValueForKey("DAYS");
181                         }
182                         else
183                             if (iTimeAgo < (3600 * 60 * 30 * 12)) {
184                                 if (Math.floor(iTimeAgo / (3600 * 60 * 30)) === 1) {
185                                     return Math.floor(iTimeAgo / (3600 * 60 * 30)) + " "+sakai.api.i18n.General.getValueForKey("MONTH");
186                                 }
187                                 return Math.floor(iTimeAgo / (3600 * 60 * 30)) + " "+sakai.api.i18n.General.getValueForKey("MONTHS");
188                             }
189                             else {
190                                 if (Math.floor(iTimeAgo / (3600 * 60 * 30 * 12) === 1)) {
191                                     return Math.floor(iTimeAgo / (3600 * 60 * 30 * 12)) + " "+sakai.api.i18n.General.getValueForKey("YEAR");
192                                 }
193                                 return Math.floor(iTimeAgo / (3600 * 60 * 30 * 12)) + " "+sakai.api.i18n.General.getValueForKey("YEARS");
194                             }
195         }
196 
197         return null;
198 
199     };

201     /**
202      * Converts all HTML to flat text and converts \n to <br />
203      * @param {String} str
204      */
205     var tidyInput = function(str){
206         str = str.toString(); // in the event its not already a string, make it one
207         str = str.replace(/&/g, "&").replace(/</g, "<").replace(/>/g, ">");
208         str = str.replace(/\n/g, '<br />');
209         return str;
210     };

216     /**
217      * Show the comments in a paged state or not
218      */
219     var displayCommentsPagedOrNot = function(){
220         jsonDisplay = {
221             "comments": [],
222             "settings": widgetSettings
223         };
224         // Loops through all the comments and does the necessary changes to render the JSON-object
225         for (var i = 0; i < json.results.length; i++) {
226             jsonDisplay.comments[i] = {};
227             var comment = json.results[i].post;
228             // Checks if the date is already parsed to a date object
229             // TODO: Get jcr:created
230             var tempDate = comment["sakai:created"];
231             try {
232                 // if the date is not a string this should generate en exception
233                 comment.date = parseDate(tempDate);
234             }
235             catch (ex) {
236                 comment.date = tempDate;
237             }
238 
239             comment.timeAgo = "about " + getTimeAgo(comment.date) + " "+sakai.api.i18n.General.getValueForKey("AGO");
240             comment.formatDate = sakai.api.l10n.transformDateTimeShort(comment.date);
241             comment.messageTxt = comment["sakai:body"];
242             comment.message = tidyInput(comment["sakai:body"]);
243             // weird json bug.
244             comment["sakai:deleted"] = (comment["sakai:deleted"] && (comment["sakai:deleted"] === "true" || comment["sakai:deleted"] === true)) ? true : false;
245 
246 
247             var user = {};
248             // User
249             // Puts the userinformation in a better structure for trimpath
250             // if (comment.profile["sling:resourceType"] === "sakai/user-profile") { // no longer in use, it seems
251             if (comment.profile) {
252                 var profile = comment.profile[0];
253                 user.fullName = sakai.api.User.getDisplayName(profile);
254                 user.picture = sakai.config.URL.USER_DEFAULT_ICON_URL;
255                 // Check if the user has a picture
256                 if (profile.picture && $.parseJSON(profile.picture).name) {
257                     user.picture = "/~" + profile["userid"] + "/public/profile/" + $.parseJSON(profile.picture).name;
258                 }
259                 user.uid = profile["userid"];
260                 user.profile = "/~" + user.uid;
261             }
262             else {
263                 // This is an anonymous user.
264                 comment.profile = {};
265                 comment.profile.fullName = "Anonymous";
266                 comment.profile.email = "noreply@sakaiproject.org";
267                 if (widgetSettings["sakai:forcename"] === true) {
268                     comment.profile.fullName = comment['sakai:name'];
269                 }
270                 if (widgetSettings["sakai:forcemail"] === true) {
271                     comment.profile.email = comment['sakai:email'];
272                 }
273             }
274 
275             comment.user = user;
276 
277             jsonDisplay.comments[i] = comment;
278         }
279         $(commentsShowComments, rootel).html($.TemplateRenderer(commentsShowCommentsTemplate, jsonDisplay));
280     };

282     /**
283      * Show all the posted comments
284      * This function first retrieves all the users who have posted in this widget and then call the displayCommentsPagedOrNot function
285      */
286     var showComments = function(){
287         // Show the nr of comments we are showing.
288         var showingComments = json.total;
289         if (widgetSettings.perPage < json.total) {
290             showingComments = widgetSettings.perPage;
291         }
292         $(commentsNumCommentsDisplayed, rootel).html(showingComments);
293         // Puts the number of comments on the page
294         $(commentsNumComments, rootel).html(json.total);
295         // Change to "comment" or "comments"
296         if (json.total === 1) {
297             $(commentsCommentComments, rootel).text(sakai.api.i18n.General.getValueForKey("COMMENT"));
298         }
299 
300 
301         // Change the page-number on the display
302         $(commentsPager, rootel).pager({
303             pagenumber: clickedPage,
304             pagecount: Math.ceil(json.total / widgetSettings.perPage),
305             buttonClickCallback: pagerClickHandler
306         });
307 
308         if (json.total > widgetSettings.perPage) {
309             $(commentsPager, rootel).show();
310         }
311         // Checks if the comments undefined or if it's length is 0
312         displayCommentsPagedOrNot();
313     };

315     /**
316      * Gets the comments from the service.
317      */
318     var getComments = function(){
319         var sortOn = "sakai:created";
320         var sortOrder = "descending";
321         var items = 10;
322         if (widgetSettings.direction && widgetSettings.direction === "comments_FirstDown") {
323             sortOrder = "ascending";
324         }
325         if (widgetSettings.perPage) {
326             items = widgetSettings.perPage;
327         }
328 
329         var url = "/var/search/comments/flat.json?sortOn=" + sortOn + "&sortOrder=" + sortOrder + "&page=" + (clickedPage - 1) + "&items=" + items + "&marker=" + tuid + "&path=" + store;
330         $.ajax({
331             url: url,
332             cache: false,
333             success: function(data){
334                 json = $.extend(data, {}, true);
335                 showComments();
336             },
337             error: function(xhr, textStatus, thrownError){
338                 alert("comments: An error occured while receiving the comments (" + xhr.status + ")");
339             }
340         });
341     };

359     /**
360      * Post a new comment
361      */
362     var postComment = function(){
363         var comment = {
364             // Replaces the \n (enters) with <br />
365             "message": $(commentsMessageTxt, rootel).val()
366         };
367         comment["sakai:type"] = "comment";
368 
369         var isLoggedIn = (me.user.anon && me.user.anon === true) ? false : true;
370         var allowPost = true;
371         // If the user is not loggedin but we allow anon comments, we check some extra fields.
372         if (!isLoggedIn && widgetSettings['sakai:allowanonymous'] === true) {
373             if (!isLoggedIn && widgetSettings['sakai:forcename']) {
374                 comment["sakai:name"] = $(commentsNamePosterTxt, rootel).val();
375                 if (comment["sakai:name"].replace(/\s/g, "") === "") {
376                     allowPost = false;
377                 }
378             }
379             if (!isLoggedIn && widgetSettings['sakai:forcemail']) {
380                 comment["sakai:email"] = $(commentsMailPosterTxt, rootel).val();
381                 if (comment["sakai:email"].replace(/\s/g, "") === "") {
382                     allowPost = false;
383                 }
384             }
385         }
386         if (!isLoggedIn && widgetSettings['sakai:allowanonymous'] === false) {
387             // This should not even happen.. Somebody is tinkering with the HTML.
388             allowPost = false;
389             alert("Anonymous users are not allowed to post comments. Please register or log in to add your comment.");
390         }
391 
392         var subject = 'Comment on /~' + currentSite;
393         var to = "internal:w-" + widgeturl + "/message";
394 
395         if (allowPost) {
396             var body = $(commentsMessageTxt, rootel).val();
397             var message = {
398                 "sakai:type": "comment",
399                 "sakai:to": to,
400                 "sakai:marker": tuid,
401                 "sakai:subject": subject,
402                 "sakai:body": body,
403                 "sakai:messagebox": "outbox",
404                 "sakai:sendstate": "pending",
405                 "_charset_":"utf-8"
406             };
407 
408 
409             var url = widgeturl + "/message.create.html";
410             $.ajax({
411                 url: url,
412                 type: "POST",
413                 cache: false,
414                 success: function(data){
415                     // Hide the form.
416                     $(commentsFillInComment, rootel).hide();
417                     // Clear the textboxes.
418                     $(commentsMessageTxt, rootel).val("");
419                     $(commentsNamePosterTxt, rootel).val("");
420                     $(commentsMailPosterTxt, rootel).val("");
421                     // Get the comments.
422                     getComments();
423                 },
424                 error: function(xhr, textStatus, thrownError){
425                     if (xhr.status === 401) {
426                         alert("You are not allowed to add comments.");
427                     }
428                     else {
429                         alert("Failed to save.");
430                     }
431                 },
432                 data: message
433             });
434         }
435         else {
436             alert("Please fill in all the fields.");
437         }
438     };

483     /**
484      * fills up the settings JSON-object
485      * @return {Object} the settings JSON-object, returns {Boolean} false if input is invalid
486      */
487     var getCommentsSettings = function(){
488         var comments = {};
489         comments.comments = [];
490 
491         // Checks if there's already some comments placed on the widget
492         comments.comments = json.comments || [];
493 
494         comments.perPage = parseInt($(commentsPageTxt, rootel).val(), 10);
495         if (isNaN(comments.perPage)) {
496             comments.perPage = defaultPostsPerPage;
497         }
498 
499         if (comments.perPage < 1) {
500             alert("Please fill in a number bigger then 0.");
501             return false;
502         }
503         // Check if a valid number is inserted
504         else
505             if ($(commentsPageTxt, rootel).val().search(/^\d*$/)) {
506                 alert("Please fill in a valid number.");
507                 return false;
508             }
509 
510 
511         comments.direction = $("input[name=" + commentsDirectionRbt + " ]:checked", rootel).val();
512 
513         // These properties are noy yet used in the comments-widget, but are saved in JCR
514         comments['sakai:allowanonymous'] = true;
515         if ($("#comments_RequireLogInID", rootel).is(":checked")) {
516             comments['sakai:allowanonymous'] = false;
517         }
518         comments['sakai:forcename'] = $(commentsNameReqChk, rootel).attr("checked");
519         comments['sakai:forcemail'] = $(commentsEmailReqChk, rootel).attr("checked");
520         comments['sakai:notification'] = $(commentsSendMailChk, rootel).attr("checked");
521         comments['sakai:notificationaddress'] = me.user.userid;
522         comments['sling:resourceType'] = 'sakai/settings';
523         comments['sakai:marker'] = tuid;
524         comments['sakai:type'] = "comment";
525 
526         return comments;
527     };

540     /**
541      * Gets the widget settings and shows the appropriate view.
542      */
543     var getWidgetSettings = function(){
544 
545         sakai.api.Widgets.loadWidgetData(tuid, function(success, data){
546             if (success) {
547                 if (!data.message) {
548                     sakai.api.Widgets.saveWidgetData(tuid, {"message":{"sling:resourceType":"sakai/messagestore"}}, null);
549                 }
550                 widgetSettings = data;
551                 // Clean up some values so that true is really true and not "true" or 1 ...
552                 var keysToClean = ['sakai:forcename', 'sakai:forcemail', 'notification', 'sakai:allowanonymous'];
553                 cleanBooleanSettings(keysToClean);
554 
555                 var isLoggedIn = (me.user.anon && me.user.anon === true) ? false : true;
556                 if (widgetSettings["sakai:allowanonymous"] === false && !isLoggedIn) {
557                     $(commentsCommentBtn, rootel).parent().hide();
558                 }
559 
560                 if (showSettings) {
561                     showSettingScreen(true, data);
562                 }
563                 else {
564                     pagerClickHandler(1);
565                 }
566             }
567             else {
568                 if (showSettings) {
569                     showSettingScreen(false, data);
570                 }
571                 else {
572                     pagerClickHandler(1);
573                 }
574             }
575         });
576
