 19 /*global $ */
 20 
 21 var sakai = sakai || {};
 22 
 23 /**
 24  * @name sakai.mycontacts
 25  *
 26  * @class mycontacts
 27  *
 28  * @description
 29  * Initialize the mycontacts widget
 30  *
 31  * @version 0.0.1
 32  * @param {String} tuid Unique id of the widget
 33  * @param {Boolean} showSettings Show the settings of the widget or not
 34  */
 35 sakai.mycontacts = function(tuid,showSettings){
 36 
 37     /////////////////////////////
 38     // Configuration variables //
 39     /////////////////////////////
 40 
 41     var rootel = $("#" + tuid);
 42     var numberFriends = 5; // The number of contacts that will be shown
 43 
 44     // - ID
 45     var mycontacts = "#mycontacts";
 46 
 47     // Contact request
 48     var mycontactsRequests = mycontacts + "_requests";
 49 
 50     // Error
 51     var mycontactsError = mycontacts + "_error";
 52     var mycontactsErrorContactserver = mycontactsError + "_contactserver";
 53 
 54     // List
 55     var mycontactsList = mycontacts + "_list";
 56 
 57     // Templates
 58     var mycontactsListTemplate = "mycontacts_list_template";
 59     var mycontactsRequestsTemplate = "mycontacts_requests_template";
 60 
 61 
 62     ///////////////////////
 63     // Utility functions //
 64     ///////////////////////
 65 
 66     /**
 67      * Parse the name for a user
 68      * @param {String} uuid Uuid of the user
 69      * @param {String} firstName Firstname of the user
 70      * @param {String} lastName Lastname of the user
 71      */
 72     var parseName = function(uuid, profile){
 73         var displayName = sakai.api.User.getDisplayName(profile);
 74         if (displayName) {
 75             return displayName;
 76         }
 77         else {
 78             return uuid;
 79         }
 80     };
 81 
 82     /**
 83      * Parse the picture for a user
 84      * @param {String} picture The picture path for a user
 85      * @param {String} userStoragePrefix The user's storage prefix
 86      */
 87     var parsePicture = function(profile, uuid){
 88         // Check if the picture is undefined or not
 89         // The picture will be undefined if the other user is in process of
 90         // changing his/her picture
 91         if (profile && profile.picture && $.parseJSON(profile.picture).name) {
 92             return "/~" + profile["rep:userId"] + "/public/profile/" + $.parseJSON(profile.picture).name;
 93         }
 94         return sakai.config.URL.USER_DEFAULT_ICON_URL;
 95     };
 96 
 97 
 98     ///////////////////////////
 99     // Get & process contacts //
100     ///////////////////////////
101 
102     /**
103      * Process the information for each friend
104      * @param {Object} contacts JSON object containing all the contacts the current user
105      */
106     var doProcessing = function(contacts){
107         var jsonFriends = {};
108 
109         // Array that will contain a specified number of contacts of the current user
110         jsonFriends.items = [];
111 
112         if (contacts.results) {
113             // Run process each friend
114             for (var i = 0, j = contacts.results.length; i < j; i++) {
115                 if (i <= numberFriends) {
116                     var friend = contacts.results[i];
117 
118                     // Set the id of the friend
119                     friend.id = friend.target;
120 
121                     // Parse the name of the friend
122                     friend.name = parseName(friend.target, friend.profile);
123 
124                     // Parse the picture of the friend
125                     friend.photo = parsePicture(friend.profile, friend.target);
126 
127                     // Add the friend to the array
128                     jsonFriends.items.push(friend);
129                 }
130             }
131         }
132         // Render the template with the contacts
133         $(mycontactsList).html($.TemplateRenderer(mycontactsListTemplate, jsonFriends));
134     };
135 
136 
137     /**
138      * Get all the contacts for the current user.
139      * It only gets the contacts that have an accepted status
140      * and the request is ordered by the first and last name of the contacts
141      */
142     var getFriends = function(){
143         $.ajax({
144             url: sakai.config.URL.CONTACTS_ACCEPTED + "?page=0&items=6",
145             cache: false,
146             success: function(data){
147 
148                 // Process the contacts: username, picture, ...
149                 doProcessing(data);
150             },
151             error: function(xhr, textStatus, thrownError) {
152 
153                 // Show the contact error
154                 $(mycontactsErrorContactserver, rootel).show();
155             }
156         });
157     };
158 
159 
160     //////////////////////
161     // Contact requests //
162     //////////////////////
163 
164     /**
165      * Get all the contact request for the current user and show
166      * them on the page.
167      */
168     var getContactRequests = function(){
169         $(mycontactsRequests).html($.TemplateRenderer(mycontactsRequestsTemplate, sakai.data.me.contacts));
170     };
171 
172 
173     /////////////////////////////
174     // Initialisation function //
175     /////////////////////////////
176 
177     var doInit = function() {
178 
179         // Get the firends for the current user
180         getFriends();
181 
182         // Get the contacts requests for the current user
183         getContactRequests();
184     };
185 
186     doInit();
187 
188 };
189 
190 sakai.api.Widgets.widgetLoader.informOnLoad("mycontacts");
