 82     // see: http://www.ietf.org/rfc/rfc2396.txt Appendix B
 83     var urlRegExp = new RegExp("^(([^:/?#]+):)?(//([^/?#]*))?([^?#]*)(\\?([^#]*))?(#(.*))?");

 85     ///////////////////////
 86     // Utility functions //
 87     ///////////////////////

 89     /**
 90      * Check if the value is a decimal or not
 91      * @param {Object} value Value that needs to be checked
 92      * @return {Boolean}
 93      *     true: is a decimal
 94      *     false: is not a decimal
 95      */
 96     var isDecimal = function(value){
 97         return (/^\d+$/).test(value);
 98     };
 99 
100     /**
101      * Check if the input url is in fact an url or not
102      * @param {String} url Url that needs to be tested
103      * @return {Boolean}
104      *     true: is an url
105      *     false: is not an url
106      */
107     var isUrl = function(url){
108         var matches = urlRegExp.exec(url);
109         // e.g. if("http:" && "localhost")
110         if(matches[1] && matches[4]) {
111             return true;
112         } else {
113             return false;
114         }
115     };
116 
117     /**
118      * Check to see if both URLs are in the same origin. See: http://en.wikipedia.org/wiki/Same_origin_policy.
119      * @param {String} url1
120      * @param {String} url2
121      * @return {Boolean}
122      *     true: in the same origin policy
123      *     false: NOT in the same origin policy
124      */
125     var isSameOriginPolicy = function(url1, url2){
126         if(url1 == url2) {
127             return true;
128         }
129         // console.log(isUrl(url1) + ": " + url1 + "=" + urlRegExp.exec(url1)[4]);
130         // console.log(isUrl(url2) + ": " + url2 + "=" + urlRegExp.exec(url2)[4]);
131         // i.e. protocol, domain (and optional port numbers) must match
132         if((urlRegExp.exec(url1)[2] == urlRegExp.exec(url2)[2]) &&
133            (urlRegExp.exec(url1)[4] == urlRegExp.exec(url2)[4])){
134             return true;
135         } else {
136             return false;
137         }
138     }

141      * Called when the data has been saved to the JCR.
142      */
143     var savedDataToJCR = function(data, textStatus, XMLHttpRequest){
144         sakai.api.Widgets.Container.informFinish(tuid, "basiclti");
145     };

152     /**
153      * Render the iframe for the widget in settings mode
154      * @param {Boolean} complete Render the preview completely or only adjust values
155      */
156     var renderIframeSettings = function(complete){
157         if (complete) {
158             json.launchDataUrl = sakai.config.URL.SDATA_FETCH_URL.replace(/__PLACEMENT__/, sakai.site.currentsite.id + "/_widgets").replace(/__TUID__/, tuid).replace(/__NAME__/, "basiclti") + '.launch.html';
159             $(basicltiSettingsPreview).html($.TemplateRenderer($basicltiSettingsPreviewTemplate, json));
160         }
161         else {
162             $(basicltiSettingsPreviewFrame).attr("style", "border: " + json.border_size + "px #" + json.border_color + " solid");
163         }
164     };
165 
166     /**
167      * Render the iframe for the widget
168      */
169     var renderIframe = function(){
170         if (json) {
171             json.launchDataUrl = sakai.config.URL.SDATA_FETCH_URL.replace(/__PLACEMENT__/, sakai.site.currentsite.id + "/_widgets").replace(/__TUID__/, tuid).replace(/__NAME__/, "basiclti") + '.launch.html';
172             json.tuidFrame = basicltiSettingsPreviewId;
173             $(basicltiMainContainer, rootel).html($.TemplateRenderer($basicltiSettingsPreviewTemplate, json));
174             // SAKIII-542 Basic LTI no longer renders IFRAME content (workaround)
175             $("#" + json.tuidFrame).attr("src", json.launchDataUrl);
176             // resize the iframe to match inner body height if in the same origin (i.e. same protocol/domain/port)
177             if(isSameOriginPolicy(window.location.href, json.ltiurl)) {
178                 $(basicltiSettingsPreviewFrame).load(function() {
179                     $(this).height($(this).contents().find("body").height() + 15); // add 10px for IE and 5px more for Gradebook weirdness
180                 });
181             }
182 
183             // SAKIII-314 We need to show the container, otherwise the second item won't be shown.
184             $(basicltiMainContainer, rootel).show();
185         }
186     };
187 
188     /**
189      * Render the html of the basicltisettings
190      */
191     var renderRemoteContentSettings = function(){
192         if (json) {
193             $(basicltiSettings).html($.TemplateRenderer($basicltiSettingsTemplate, json));
194         }
195     };
196 
197     /**
198      * Render the color container
199      */
200     var renderColorContainer = function(){
201         if (json) {
202             $(basicltiSettingsColorContainer).html($.TemplateRenderer($basicltiSettingsColorContainerTemplate, json));
203         }
204     };

211     /**
212      * Display the iframe in normal mode
213      * @param {Object} parameters JSON object that contains the necessary information for the iframe
214      */
215     var displayRemoteContent = function(parameters){
216         // default to some reasonable vaules if the settings node does not have them (maybe a legacy node)
217         if (parameters.border_size == null) {
218             parameters.border_size = 0;
219         }
220         if (parameters.border_color == null) {
221             parameters.border_color = "cccccc";
222         }
223         if (parameters.width == null) {
224             parameters.width = defaultWidth;
225         }
226         if (parameters.width_unit == null) {
227             parameters.width_unit = defaultWidthUnit;
228         }
229         if (parameters.frame_height == null) {
230             parameters.frame_height = defaultHeight;
231         }
232         json = parameters;
233         renderIframe();
234     };
235 
236     /**
237      * Save the basiclti to the jcr
238      */
239     var saveRemoteContent = function(){
240         if (json.ltiurl !== "") {
241             json[":operation"] = "basiclti";
242             json["sling:resourceType"] = "sakai/basiclti";
243             json.ltiurl = $(basicltiSettingsLtiUrl).val() || "";
244             json.ltikey = $(basicltiSettingsLtiKey).val() || "";
245             json.ltisecret = $(basicltiSettingsLtiSecret).val() || "";
246             json["debug@TypeHint"] = "Boolean";
247             json.debug = $('#basiclti_settings_debug:checked').val() != null;
248             json["release_names@TypeHint"] = "Boolean";
249             json.release_names = $('#basiclti_settings_release_names:checked').val() != null;
250             json["release_principal_name@TypeHint"] = "Boolean";
251             json.release_principal_name = $('#basiclti_settings_release_principal_name:checked').val() != null;
252             json["release_email@TypeHint"] = "Boolean";
253             json.release_email = $('#basiclti_settings_release_email:checked').val() != null;
254             json.launchDataUrl = ""; // does not need to be persisted
255             json.tuidFrame = ""; // does not need to be persisted
256             json.defined = ""; // what the heck is this? Where does it come from?
257             json._MODIFIERS = null; // trimpath garbage - probably need a more selective way of saving data
258             var saveUrl = sakai.config.URL.SDATA_FETCH_URL.replace(/__PLACEMENT__/, sakai.site.currentsite.id + "/_widgets").replace(/__TUID__/, tuid).replace(/__NAME__/, "basiclti");
259             $.ajax({
260                 type: 'POST',
261                 url: saveUrl,
262                 data: json,
263                 success: savedDataToJCR
264             });
265         }
266         else {
267             alert("Please specify a URL");
268         }
269     };

271     /**
272      * Change the direction (up/down) of the arrow for the advanced settings
273      */
274     var changeAdvancedSettingsArrow = function(){
275         if (isAdvancedSettingsVisible) {
276             $(basicltiSettingsAdvancedDown, rootel).hide();
277             $(basicltiSettingsAdvancedUp, rootel).show();
278         }
279         else {
280             $(basicltiSettingsAdvancedUp, rootel).hide();
281             $(basicltiSettingsAdvancedDown, rootel).show();
282         }
283     };

290     /*
291      * Add binding to the color boxes
292      */
293     var addColorBinding = function(){
294         $(".basiclti_settings_color").click(function(){
295             json.border_color = $(this).attr("id").split("_")[$(this).attr("id").split("_").length - 1];
296             renderIframeSettings(false);
297             renderColorContainer();
298             addColorBinding();
299         });
300     };
301 
302     /*
303      * Add binding to all the elements
304      */
305     var addBinding = function(){
306 
307         // Change the url for the iFrame
308         $(basicltiSettingsLtiUrl).change(function(){
309             var urlValue = $(this).val();
310             if (urlValue !== "") {
311                 // Check if someone already wrote http inside the url
312                 if (!isUrl(urlValue)) {
313                     urlValue = 'http://' + urlValue;
314                 }
315                 json.ltiurl = urlValue;
316                 //renderIframeSettings(true); // LDS disabled preview
317             }
318         });
319 
320         // Change the iframe width
321         $(basicltiSettingsWidth).change(function(){
322             var widthValue = $(basicltiSettingsWidth).val();
323 
324             if (isDecimal(widthValue)) {
325                 json.width = widthValue;
326             }
327             renderIframeSettings(false);
328         });
329 
330         // Change the iframe height
331         $(basicltiSettingsHeight).change(function(){
332             var heightValue = $(basicltiSettingsHeight).val();
333 
334             if (isDecimal(heightValue)) {
335                 json.frame_height = heightValue;
336             }
337             renderIframeSettings(false);
338         });
339 
340         // Change the border width
341         $(basicltiSettingsBorders).change(function(){
342             var borderValue = $(basicltiSettingsBorders).val();
343             if (isDecimal(borderValue)) {
344                 json.border_size = borderValue;
345                 renderIframeSettings(false);
346             }
347         });
348 
349         // Toggle the advanced view
350         $(basicltiSettingsAdvancedToggleSettings).click(function(){
351             $("#basiclti_settings_advanced", rootel).toggle();
352             isAdvancedSettingsVisible = !isAdvancedSettingsVisible;
353             changeAdvancedSettingsArrow();
354         });
355 
356         // When you click on one of the width units (px or percentage)
357         $(basicltiSettingsWidthUnitClass).click(function(){
358             var widthUnitValue = $(this).attr("id").split("_")[$(this).attr("id").split("_").length - 1];
359             if (widthUnitValue === "px") {
360                 json.width_unit = widthUnitValue;
361             }
362             else {
363                 json.width_unit = "%";
364             }
365             $(basicltiSettingsWidthUnitClass).removeClass(basicltiSettingsWidthUnitSelectedClass);
366             $(this).addClass(basicltiSettingsWidthUnitSelectedClass);
367             renderIframeSettings(false);
368         });
369 
370         // When you push the save button..
371         $(basicltiSettingsInsert).click(function(){
372             saveRemoteContent();
373         });
374 
375         // Cancel it
376         $(basicltiSettingsCancel).click(function(){
377             sakai.api.Widgets.Container.informCancel(tuid, "basiclti");
378         });
379 
380         addColorBinding();
381     };

388     /**
389      * Function that fills in the input fields in the settings tab.
390      * @param {Object} parameters A JSON object that contains the necessary information.
391      * @param {Boolean} exists Does there exist a previous basiclti
392      */
393     var displaySettings = function(parameters, exists){
394         if (exists && parameters.ltiurl) {
395             json = parameters;
396         }
397         else { // use default values
398             json = {
399                 ltiurl: "",
400                 ltikey: "",
401                 ltisecret: "",
402                 release_names: true,
403                 release_principal_name: true,
404                 release_email: true,
405                 border_size: 0,
406                 border_color: "cccccc",
407                 frame_height: defaultHeight,
408                 width: defaultWidth,
409                 width_unit: defaultWidthUnit
410             };
411         }
412         renderRemoteContentSettings();
413         //renderIframeSettings(true); // LDS disabled preview
414         renderColorContainer();
415         addBinding(); // Add binding to the various elements
416         changeAdvancedSettingsArrow();
417         $(basicltiSettings).show(); // Show the basiclti settings
418     };
419 
420     /*
421      * Is the widget in settings mode or not
422      */
423     if (showSettings) {
424         $(basicltiMainContainer).hide();
425         $(basicltiSettings).show();
426     }
427     else {
428         $(basicltiSettings).hide();
429         $(basicltiMainContainer).show();
430     }
431 
432     /**
433      * Will fetch the URL and other parameters from the JCR and according to which
434      * view we are in, fill in the settings or display an iframe.
435      */
436     var getRemoteContent = function(){
437         var settingsUrl = sakai.config.URL.SDATA_FETCH_URL.replace(/__PLACEMENT__/, sakai.site.currentsite.id + "/_widgets").replace(/__TUID__/, tuid).replace(/__NAME__/, "basiclti");
438         $.ajax({
439             url: settingsUrl,
440             cache: false,
441             success: function(data){
442 
443                 if (showSettings) {
444                     displaySettings(data, true); // Fill in the settings page.
445                 }
446                 else {
447                     displayRemoteContent(data); // Show the frame
448                 }
449             },
450             error: function(xhr, textStatus, thrownError){
451                 // When the request isn't successful, it means that  there was no existing basiclti
452                 // so we show the basic settings.
453                 displaySettings(null, false);
454             }
455         });
456     };
457 
458     getRemoteContent();
459 };
